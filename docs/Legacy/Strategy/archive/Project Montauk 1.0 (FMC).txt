//@version=5
strategy("FMC 1.0", overlay=true)

// === INPUTS ===

// MACD Settings
fast_length   = input.int(29,  "MACD Fast Length",   minval=1)
slow_length   = input.int(89,  "MACD Slow Length",   minval=1)
signal_length = input.int(10,  "MACD Signal Length", minval=1)

// ATR Stop Settings
atr_period = input.int(14,  "ATR Period",      minval=1)
atr_mult   = input.float(3, "ATR Mult (Stop)", step=0.1)

// Cooldown Settings
cooldown_period = input.int(3, "Cooldown (Days)", minval=0)

// SMA Filter Setting
trend_sma_length = input.int(200, "SMA Length (Days)", minval=1)

// Slope-Based No-Trade Zone Settings
slope_lookback       = input.int(10,  "Slope Lookback Days",       minval=1)
rolling_slope_length = input.int(10,  "Rolling Slope Length",      minval=1)
slope_threshold      = input.float(0.1, "Slope Threshold", step=0.01, tooltip="Minimum absolute slope to consider the market as trending.")

// === CALCULATIONS ===

// 1) MACD
[macd_line, signal_line, _] = ta.macd(close, fast_length, slow_length, signal_length)

// 2) ATR for Volatility Stop
atr_value = ta.atr(atr_period)

// 3) Trend SMA Calculation
trend_sma = ta.sma(close, trend_sma_length)
bool above_trend_sma = close > trend_sma

// 4) Highest price since entry (for ATR-based trailing stop)
var float highest_price_since_entry = na
if strategy.position_size > 0
    if na(highest_price_since_entry)
        highest_price_since_entry := close
    else
        highest_price_since_entry := math.max(highest_price_since_entry, close)
else
    highest_price_since_entry := na

// 5) ATR stop level
float atr_stop_level = highest_price_since_entry - (atr_mult * atr_value)

// === STATE & LOGIC ===
var int last_stop_bar = na
bool in_cooldown      = (bar_index - last_stop_bar) <= cooldown_period

// --- Slope Calculation ---
float sma_past  = ta.sma(close, trend_sma_length)[slope_lookback]
float slope_val = trend_sma - sma_past

// Rolling Average of Slope
float avg_slope = ta.sma(slope_val, rolling_slope_length)

// No-Trade Zone Condition (always on)
bool no_trade_zone = math.abs(avg_slope) < slope_threshold

// === Priority #1: Volatility Stop (ATR)
bool volatility_stop_trigger = (strategy.position_size > 0) and (close < atr_stop_level)

// === MACD Conditions (Only if not overridden by No-Trade Zone)
bool bullish_macd  = (macd_line > signal_line + 0.5)
bool bearish_macd  = (macd_line < signal_line - 0.5)

// Buy Condition => Must not be in no-trade zone
bool is_bullish    = bullish_macd and above_trend_sma and not in_cooldown and not no_trade_zone
bool buy_condition = is_bullish and (strategy.position_size == 0)

// Sell Condition => Must not be in no-trade zone (unless ATR stop triggers)
bool macd_sell_condition = bearish_macd and not no_trade_zone

// If any sell triggers, reset cooldown
if volatility_stop_trigger or macd_sell_condition
    last_stop_bar := bar_index

// === TRADING LOGIC ===

// Priority #1: ATR stop exit always overrides
if (volatility_stop_trigger) and (strategy.position_size > 0)
    strategy.close("Buy")

// Priority #2/3: MACD exit if not in no-trade zone
else if (macd_sell_condition) and (strategy.position_size > 0)
    strategy.close("Buy")

// Buying logic (only if not in no-trade zone)
if buy_condition
    strategy.entry("Buy", strategy.long, qty=100)

// === PLOTTING ===

// 1) Price with position color
plot(close, color = strategy.position_size > 0 ? color.green : color.red, linewidth=2, title="Price with Position Color")

// 2) Trend SMA line
plot(trend_sma, color=color.blue, linewidth=2, title="Trend SMA")

// 3) MACD & Signal lines (overlayed)
plot(macd_line,   color=color.green, linewidth=1, title="MACD")
plot(signal_line, color=color.red,   linewidth=1, title="Signal")

// Rolling Avg Slope
plot(avg_slope, color=color.orange, linewidth=1, title="Rolling Avg Slope")
hline(slope_threshold,  "Slope+Threshold", color=color.gray, linestyle=hline.style_dotted)
hline(-slope_threshold, "Slope-Threshold", color=color.gray, linestyle=hline.style_dotted)

// Background Shading for No-Trade Zone
bgcolor(no_trade_zone ? color.new(color.red, 90) : na, title="No-Trade Zone Background")

// Plot Shapes for Entering and Exiting No-Trade Zone
bool no_trade_zone_prev = ta.valuewhen(bar_index > 0, no_trade_zone[1], 1)
bool enter_no_trade     = (no_trade_zone and not no_trade_zone_prev)
bool exit_no_trade      = (not no_trade_zone and no_trade_zone_prev)

plotshape(enter_no_trade, title="Enter No-Trade Zone", location=location.abovebar, color=color.red,   style=shape.labeldown, text="No Trade")
plotshape(exit_no_trade,  title="Exit No-Trade Zone",  location=location.belowbar, color=color.green, style=shape.labelup,   text="Trade")

// Alert Conditions for No-Trade Zone
alertcondition(enter_no_trade, title="Enter No-Trade Zone", message="Strategy has entered the No-Trade Zone (flat slope).")
alertcondition(exit_no_trade,  title="Exit No-Trade Zone",  message="Strategy has exited the No-Trade Zone. Trading allowed.")
