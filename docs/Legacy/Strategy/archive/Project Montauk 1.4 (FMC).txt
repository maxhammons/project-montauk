//@version=5
strategy("FMC 1.4 – unified overlay", overlay=true)

// === INPUTS ===
// MACD
fast_length     = input.int(29,  "MACD Fast Length")
slow_length     = input.int(89,  "MACD Slow Length")
signal_length   = input.int(10,  "MACD Signal Length")
buffer_pct      = input.float(0.001, "MACD Buffer (% of price)", step=0.0001)

// ATR trailing stop
atr_period      = input.int(14,  "ATR Period")
atr_mult        = input.float(3,   "ATR Mult (Stop)", step=0.1)

// Cool-down
cooldown_period = input.int(3,   "Cooldown (Bars)")

// 200-SMA filter
trend_sma_len   = input.int(200, "200-SMA Length")

// Flat-trend guard
slope_lookback  = input.int(10,  "Slope Lookback")
rolling_slope   = input.int(10,  "Slope Averaging Len")
slope_thresh    = input.float(0.0007, "Flat-Trend Threshold")

// Short EMA for exit filter
short_ma_len    = input.int(21,  "Exit EMA Length")
ma_buf_mult     = input.float(0.5,"EMA Buffer × ATR", step=0.1)

// Order size
order_size      = input.int(100, "Position Size")

// === CALCULATIONS ===
// MACD + buffer
[macd_line, sig_line, _] = ta.macd(close, fast_length, slow_length, signal_length)
macd_diff    = macd_line - sig_line
bullish_macd = ta.crossover(macd_line, sig_line) and macd_diff > buffer_pct * close
bearish_macd = ta.crossunder(macd_line, sig_line) and macd_diff < -buffer_pct * close

// ATR stop anchor
atr_val      = ta.atr(atr_period)
var float hi = na
hi := strategy.position_size > 0
     ? nz(hi[1]) == na ? close : math.max(hi[1], close)
     : na
atr_stop     = hi - atr_mult * atr_val

// 200-SMA + slope guard
sma200       = ta.sma(close, trend_sma_len)
above200     = close > sma200
slope_pct    = (sma200 - sma200[slope_lookback]) / sma200[slope_lookback]
avg_slope    = ta.sma(slope_pct, rolling_slope)
flat_zone    = math.abs(avg_slope) < slope_thresh

// Short EMA + buffer
ema21        = ta.ema(close, short_ma_len)
ma_buf       = ma_buf_mult * atr_val
ema_down     = ema21 < ema21[1]

// Cool-down
var int last_exit = na
in_cd       = not na(last_exit) and (bar_index - last_exit <= cooldown_period)

// MACD-then-SMA state
var bool ready = false
if bullish_macd and not flat_zone and not in_cd
    ready := true
if bearish_macd or strategy.position_size > 0
    ready := false

// EXITS
vol_exit = strategy.position_size > 0 and ema_down and close < atr_stop and close < ema21 - ma_buf
macd_exit= bearish_macd and not flat_zone
if vol_exit or macd_exit
    strategy.close("Buy")
    last_exit := bar_index
    ready     := false

// ENTRIES
if bullish_macd and above200 and avg_slope > 0 and not flat_zone and not in_cd and strategy.position_size == 0
    strategy.entry("Buy", strategy.long, qty=order_size)
    ready := false

if ready and ta.crossover(close, sma200) and avg_slope > 0 and not flat_zone and not in_cd and strategy.position_size == 0
    strategy.entry("Buy", strategy.long, qty=order_size)
    ready := false

// === PLOTTING (all on main panel) ===
// 1) Price coloured by position
plot(close, color = strategy.position_size>0 ? color.green : color.red, linewidth=2, title="Price")

// 2) Trend lines
plot(sma200, color=color.blue,   linewidth=2, title="200-SMA")
plot(ema21,  color=color.yellow, linewidth=1, title="21-EMA")

// 3) MACD & Signal
plot(macd_line, color=color.green, linewidth=1, title="MACD")
plot(sig_line,  color=color.red,   linewidth=1, title="Signal")

// 4) ATR stop
plot(strategy.position_size>0 ? atr_stop : na, color=color.purple, style=plot.style_linebr, title="ATR Stop")

// 5) Slope % & thresholds
plot(avg_slope, color=color.orange, linewidth=1, title="Avg Slope %")
hline( slope_thresh,  "Slope +Thr", linestyle=hline.style_dotted)
hline(-slope_thresh,  "Slope −Thr", linestyle=hline.style_dotted)

// 6) No-trade shading
bgcolor(flat_zone ? color.new(color.red,90) : na, title="No-Trade Zone")
