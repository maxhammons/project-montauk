//@version=6
strategy("Montauk 9.0c — Regime Composite",
     overlay=true,
     shorttitle="Montauk 9.0c",
     initial_capital=1000,
     default_qty_type=strategy.percent_of_equity,
     default_qty_value=100,
     pyramiding=0,
     commission_type=strategy.commission.percent,
     commission_value=0,
     slippage=0,
     calc_on_every_tick=false,
     process_orders_on_close=true)

// ─────────────────────────────────────────────────────────────────────────────
// Strategy: Regime Composite
//
// 2-of-3 signal confirmation system. Entry requires at least min_signals of:
//   1. RSI recovering from oversold (cross up through entry threshold)
//   2. MACD histogram turning positive and rising
//   3. Volatility calming (short vol / long vol < ratio)
// All gated by trend EMA. Exit uses the same 2-of-3 logic on the sell side,
// plus an unconditional RSI panic exit.
//
// Optimizer best (2026-04-06): fitness 1.6622, 2.662x vs B&H, 46.3% CAGR,
//   100% win rate across 12 trades over 17 years.
// ─────────────────────────────────────────────────────────────────────────────

// ─────────────────────────────────────────────────────────────────────────────
// Group 1 — RSI
// ─────────────────────────────────────────────────────────────────────────────
rsiLen      = input.int(9,      title="RSI Length",              minval=1,    group="1 - RSI")
entryRsi    = input.float(40.0, title="Entry RSI Threshold",     step=1.0,    group="1 - RSI")
exitRsi     = input.float(75.0, title="Exit RSI (Overbought)",   step=1.0,    group="1 - RSI")
panicRsi    = input.float(10.0, title="Panic RSI (Always Exit)", step=1.0,    group="1 - RSI")

// ─────────────────────────────────────────────────────────────────────────────
// Group 2 — MACD
// ─────────────────────────────────────────────────────────────────────────────
macdFast    = input.int(14,     title="MACD Fast Length",        minval=1,    group="2 - MACD")
macdSlow    = input.int(36,     title="MACD Slow Length",        minval=1,    group="2 - MACD")
macdSig     = input.int(5,      title="MACD Signal Length",      minval=1,    group="2 - MACD")
macdExitThr = input.float(-4.5, title="MACD Exit Threshold",     step=0.5,    group="2 - MACD")

// ─────────────────────────────────────────────────────────────────────────────
// Group 3 — Volatility Regime
// ─────────────────────────────────────────────────────────────────────────────
volShortLen  = input.int(30,    title="Vol Short Length",         minval=2,    group="3 - Volatility Regime")
volLongLen   = input.int(90,    title="Vol Long Length",          minval=2,    group="3 - Volatility Regime")
volRatioMax  = input.float(1.0, title="Entry Vol Ratio Max",     step=0.05,   group="3 - Volatility Regime")
volExitRatio = input.float(1.1, title="Exit Vol Ratio",          step=0.05,   group="3 - Volatility Regime")

// ─────────────────────────────────────────────────────────────────────────────
// Group 4 — Trend Filter
// ─────────────────────────────────────────────────────────────────────────────
trendLen    = input.int(75,     title="Trend EMA Length",         minval=1,    group="4 - Trend Filter")

// ─────────────────────────────────────────────────────────────────────────────
// Group 5 — Signal Confirmation
// ─────────────────────────────────────────────────────────────────────────────
minSignals  = input.int(2,      title="Min Signals for Entry/Exit", minval=1, maxval=3, group="5 - Signal Confirmation")

// ─────────────────────────────────────────────────────────────────────────────
// Group 6 — Cooldown
// ─────────────────────────────────────────────────────────────────────────────
enableCooldown   = input.bool(true, title="Enable Sell Cooldown",             group="6 - Cooldown")
sellCooldownBars = input.int(10,    title="Cooldown (bars)",      minval=0,   group="6 - Cooldown")

// ─────────────────────────────────────────────────────────────────────────────
// Indicators
// ─────────────────────────────────────────────────────────────────────────────
rsi      = ta.rsi(close, rsiLen)
trendEma = ta.ema(close, trendLen)

// MACD histogram
[macdLine, signalLine, macdHist] = ta.macd(close, macdFast, macdSlow, macdSig)

// Realized volatility = stddev of daily simple returns
dailyReturn = close[1] != 0.0 ? (close - close[1]) / close[1] : 0.0
volShort    = ta.stdev(dailyReturn, volShortLen)
volLong     = ta.stdev(dailyReturn, volLongLen)
volRatio    = volLong > 0.0 ? volShort / volLong : 1.0

// ─────────────────────────────────────────────────────────────────────────────
// Entry Signal Count (2-of-3)
// ─────────────────────────────────────────────────────────────────────────────
trendOk = close > trendEma

// Signal 1: RSI recovering from low
sig1Entry = rsi[1] < entryRsi and rsi >= entryRsi

// Signal 2: MACD histogram positive and rising
sig2Entry = macdHist > macdHist[1] and macdHist > 0

// Signal 3: Vol calming (short vol < long vol * ratio)
sig3Entry = volRatio < volRatioMax

entryCount = (sig1Entry ? 1 : 0) + (sig2Entry ? 1 : 0) + (sig3Entry ? 1 : 0)
entrySignal = entryCount >= minSignals and trendOk

// ─────────────────────────────────────────────────────────────────────────────
// Exit Signal Count (2-of-3) + Panic Override
// ─────────────────────────────────────────────────────────────────────────────

// Signal 1: RSI overbought
sig1Exit = rsi >= exitRsi

// Signal 2: MACD histogram deeply negative
sig2Exit = macdHist < macdExitThr

// Signal 3: Vol spiking
sig3Exit = volRatio > volExitRatio

exitCount = (sig1Exit ? 1 : 0) + (sig2Exit ? 1 : 0) + (sig3Exit ? 1 : 0)

isCompositeExit = strategy.position_size > 0 and exitCount >= minSignals
isPanicExit     = strategy.position_size > 0 and rsi < panicRsi

// ─────────────────────────────────────────────────────────────────────────────
// Exit Reason Label
// ─────────────────────────────────────────────────────────────────────────────
var string exitReason = na
if strategy.position_size > 0
    if isPanicExit
        exitReason := "RSI Panic"
    else if isCompositeExit
        exitReason := "Composite (" + (sig1Exit ? "R" : "") + (sig2Exit ? "M" : "") + (sig3Exit ? "V" : "") + ")"

// ─────────────────────────────────────────────────────────────────────────────
// Exit Execution
// ─────────────────────────────────────────────────────────────────────────────
var int lastSellBar = na

exitCond = isPanicExit or isCompositeExit

if exitCond
    strategy.close("Long")
    lastSellBar := bar_index
    label.new(bar_index, high, exitReason,
              yloc=yloc.abovebar,
              style=label.style_label_down,
              color=isPanicExit ? color.red : color.orange,
              textcolor=color.white, size=size.tiny)

// ─────────────────────────────────────────────────────────────────────────────
// Entry with Cooldown
// ─────────────────────────────────────────────────────────────────────────────
canEnter = strategy.position_size == 0 and
     (not enableCooldown or na(lastSellBar) or (bar_index - lastSellBar) > sellCooldownBars)

if entrySignal and canEnter
    strategy.entry("Long", strategy.long)

// ─────────────────────────────────────────────────────────────────────────────
// Visuals
// ─────────────────────────────────────────────────────────────────────────────
plot(trendEma, "Trend EMA", color=color.new(color.white, 60), linewidth=1)

// Green while in position
bgcolor(strategy.position_size > 0 ? color.new(color.green, 90) : na, title="In Position")

// Red tint below trend EMA
bgcolor(close < trendEma ? color.new(color.red, 92) : na, title="Below Trend")

// Gray during cooldown
inCooldown = not na(lastSellBar) and (bar_index - lastSellBar) <= sellCooldownBars and strategy.position_size == 0
bgcolor(inCooldown ? color.new(color.gray, 90) : na, title="Cooldown")
