//@version=6
strategy("Montauk 9.0a — RSI Regime",
     overlay=true,
     shorttitle="Montauk 9.0a",
     initial_capital=1000,
     default_qty_type=strategy.percent_of_equity,
     default_qty_value=100,
     pyramiding=0,
     commission_type=strategy.commission.percent,
     commission_value=0,
     slippage=0,
     calc_on_every_tick=false,
     process_orders_on_close=true)

// ─────────────────────────────────────────────────────────────────────────────
// Strategy: RSI Regime
//
// Core idea: TECL, as a 3x leveraged ETF, experiences extreme oversold readings
// during bear markets. When RSI(10) crosses back up through 35 while price is
// above the 150-bar EMA (long-term uptrend intact), this marks the start of
// a new bull leg. Hold until RSI reaches extreme overbought (85) or panics
// back below 15. Produces ~0.7 trades/year — pure regime-level timing.
//
// All-time best params (optimizer, 2026-04-04):
//   rsi_len=13, trend_len=150, entry_rsi=35, exit_rsi=80, panic_rsi=15, cooldown=5
//   Result: 2.132x vs buy-and-hold, 44.4% CAGR, 65.8% max DD, 0.6 trades/yr
// ─────────────────────────────────────────────────────────────────────────────

// ─────────────────────────────────────────────────────────────────────────────
// Group 1 — RSI
// ─────────────────────────────────────────────────────────────────────────────
rsiLen   = input.int(13,    "RSI Length",                    minval=1,              group="1 - RSI")
entryRSI = input.float(35.0, "Entry RSI — cross above to buy", minval=1, maxval=98, step=1.0, group="1 - RSI")
exitRSI  = input.float(80.0, "Exit RSI — overbought",        minval=2, maxval=99,  step=1.0, group="1 - RSI")
panicRSI = input.float(15.0, "Panic RSI — extreme sell",     minval=1, maxval=98,  step=1.0, group="1 - RSI")

// ─────────────────────────────────────────────────────────────────────────────
// Group 2 — Trend Filter
// Price must be above this EMA to allow entry. Blocks buying into long-term
// downtrends (e.g. the 2008 crash before recovery).
// ─────────────────────────────────────────────────────────────────────────────
trendLen = input.int(150, "Trend EMA Length", minval=1, group="2 - Trend Filter")

// ─────────────────────────────────────────────────────────────────────────────
// Group 3 — Exit Cooldown
// After any exit, block re-entry for N bars to avoid whipsawing back in
// immediately after a panic sell.
// ─────────────────────────────────────────────────────────────────────────────
cooldownBars = input.int(5, "Cooldown Bars After Exit", minval=0, group="3 - Exit Cooldown")

// ─────────────────────────────────────────────────────────────────────────────
// Indicators
// ─────────────────────────────────────────────────────────────────────────────
rsiVal   = ta.rsi(close, rsiLen)
trendEMA = ta.ema(close, trendLen)

// ─────────────────────────────────────────────────────────────────────────────
// Cooldown Tracking
// ─────────────────────────────────────────────────────────────────────────────
var int lastSellBar = na

canEnter = strategy.position_size == 0 and
     (na(lastSellBar) or (bar_index - lastSellBar) > cooldownBars)

// ─────────────────────────────────────────────────────────────────────────────
// Exit Conditions
// ─────────────────────────────────────────────────────────────────────────────

// Exit 1 — RSI Overbought: RSI reaches extreme high. Primary exit — fires on
// most trades after a sustained bull run.
isOverboughtExit = strategy.position_size > 0 and rsiVal >= exitRSI

// Exit 2 — RSI Panic: RSI crashes back below extreme oversold while in position.
// Rare but important — catches sudden reversals mid-trade.
isPanicExit = strategy.position_size > 0 and rsiVal < panicRSI

// ─────────────────────────────────────────────────────────────────────────────
// Exit Execution (priority order: overbought first, then panic)
// ─────────────────────────────────────────────────────────────────────────────
var string exitReason = na

if isOverboughtExit
    exitReason := "RSI Overbought"
    strategy.close("Long")
    lastSellBar := bar_index
    label.new(bar_index, high, exitReason,
              yloc=yloc.abovebar, style=label.style_label_down,
              color=color.orange, textcolor=color.white, size=size.tiny)

else if isPanicExit
    exitReason := "RSI Panic"
    strategy.close("Long")
    lastSellBar := bar_index
    label.new(bar_index, high, exitReason,
              yloc=yloc.abovebar, style=label.style_label_down,
              color=color.red, textcolor=color.white, size=size.tiny)

// ─────────────────────────────────────────────────────────────────────────────
// Entry — RSI crosses up through entry level + price above trend EMA
// ─────────────────────────────────────────────────────────────────────────────
trendOk      = close > trendEMA
rsiBullCross = ta.crossover(rsiVal, entryRSI)

if rsiBullCross and trendOk and canEnter
    strategy.entry("Long", strategy.long)

// ─────────────────────────────────────────────────────────────────────────────
// Visuals
// ─────────────────────────────────────────────────────────────────────────────
plot(trendEMA, "Trend EMA", color=color.new(color.blue, 60), linewidth=1)

// Shade background green while in position
bgcolor(strategy.position_size > 0 ? color.new(color.green, 90) : na,
        title="In Position")

// Shade background gray during cooldown
inCooldown = not na(lastSellBar) and (bar_index - lastSellBar) <= cooldownBars and strategy.position_size == 0
bgcolor(inCooldown ? color.new(color.gray, 90) : na, title="Cooldown")
