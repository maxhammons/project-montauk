//@version=6
strategy("Montauk 9.0b — RSI Vol Regime",
     overlay=true,
     shorttitle="Montauk 9.0b",
     initial_capital=1000,
     default_qty_type=strategy.percent_of_equity,
     default_qty_value=100,
     pyramiding=0,
     commission_type=strategy.commission.percent,
     commission_value=0,
     slippage=0,
     calc_on_every_tick=false,
     process_orders_on_close=true)

// ─────────────────────────────────────────────────────────────────────────────
// Group 1 — RSI
// Entry on RSI cross-up; exit when RSI overbought or panic-low
// ─────────────────────────────────────────────────────────────────────────────
rsiLen       = input.int(7,   title="RSI Length",          minval=1,    group="1 - RSI")
entryRsi     = input.float(40.0, title="Entry RSI Threshold",  step=1.0, group="1 - RSI")
exitRsi      = input.float(85.0, title="Exit RSI (Overbought)", step=1.0, group="1 - RSI")
panicRsi     = input.float(10.0, title="Exit RSI (Panic Low)",  step=1.0, group="1 - RSI")

// ─────────────────────────────────────────────────────────────────────────────
// Group 2 — Trend Filter
// Entry only allowed when close > trend EMA (regime filter)
// ─────────────────────────────────────────────────────────────────────────────
trendLen     = input.int(100, title="Trend EMA Length",    minval=1,    group="2 - Trend Filter")

// ─────────────────────────────────────────────────────────────────────────────
// Group 3 — Volatility Regime
// Entry filter: short vol / long vol < vol_ratio_max  (storm passing)
// Exit trigger: short vol / long vol > vol_exit_ratio  (new crash starting)
// Realized vol = stddev of simple daily returns over N bars
// ─────────────────────────────────────────────────────────────────────────────
volShortLen  = input.int(25,  title="Vol Short Length",    minval=2,    group="3 - Volatility Regime")
volLongLen   = input.int(60,  title="Vol Long Length",     minval=2,    group="3 - Volatility Regime")
volRatioMax  = input.float(1.0, title="Entry Vol Ratio Max (short/long)", step=0.05, group="3 - Volatility Regime")
volExitRatio = input.float(1.6, title="Exit Vol Ratio (short/long)",      step=0.05, group="3 - Volatility Regime")

// ─────────────────────────────────────────────────────────────────────────────
// Group 4 — Sell Cooldown
// ─────────────────────────────────────────────────────────────────────────────
enableCooldown   = input.bool(true, title="Enable Sell Cooldown",                    group="4 - Sell Cooldown")
sellCooldownBars = input.int(10,    title="Sell Cooldown (bars)", minval=0,          group="4 - Sell Cooldown")

// ─────────────────────────────────────────────────────────────────────────────
// Calculations
// ─────────────────────────────────────────────────────────────────────────────
rsi      = ta.rsi(close, rsiLen)
trendEma = ta.ema(close, trendLen)

// Realized volatility = stddev(daily simple returns, N)
// daily_return[i] = (close[i] - close[i-1]) / close[i-1]
dailyReturn = close[1] != 0.0 ? (close - close[1]) / close[1] : 0.0
volShort    = ta.stdev(dailyReturn, volShortLen)
volLong     = ta.stdev(dailyReturn, volLongLen)
volRatio    = volLong > 0.0 ? volShort / volLong : 1.0

// ─────────────────────────────────────────────────────────────────────────────
// Entry Conditions
// ─────────────────────────────────────────────────────────────────────────────
trendOk     = close > trendEma
volEntryOk  = volRatio < volRatioMax
rsiCrossUp  = ta.crossover(rsi, entryRsi)

entrySignal = rsiCrossUp and trendOk and volEntryOk

// ─────────────────────────────────────────────────────────────────────────────
// Exit Conditions
// ─────────────────────────────────────────────────────────────────────────────
isRsiOverbought = rsi >= exitRsi
isRsiPanic      = rsi < panicRsi
isVolSpike      = volRatio > volExitRatio

// ─────────────────────────────────────────────────────────────────────────────
// Exit Reason  (for on-chart label)
// ─────────────────────────────────────────────────────────────────────────────
var string exitReason = na
if strategy.position_size > 0
    if isRsiOverbought
        exitReason := "RSI Overbought"
    else if isRsiPanic
        exitReason := "RSI Panic"
    else if isVolSpike
        exitReason := "Vol Spike"

// ─────────────────────────────────────────────────────────────────────────────
// Unified Exit
// ─────────────────────────────────────────────────────────────────────────────
exitCond = strategy.position_size > 0 and (isRsiOverbought or isRsiPanic or isVolSpike)

if exitCond
    strategy.close("Long")
    label.new(bar_index, high, exitReason,
              yloc=yloc.abovebar,
              style=label.style_label_down,
              color=color.red, textcolor=color.white, size=size.tiny)

// ─────────────────────────────────────────────────────────────────────────────
// Entry with Cooldown
// ─────────────────────────────────────────────────────────────────────────────
var int lastSellBar = na

if exitCond
    lastSellBar := bar_index

canEnter = strategy.position_size == 0 and
     (not enableCooldown or na(lastSellBar) or (bar_index - lastSellBar) > sellCooldownBars)

if entrySignal and canEnter
    strategy.entry("Long", strategy.long)

// ─────────────────────────────────────────────────────────────────────────────
// Background Shading
// Red tint when below trend EMA (no-trade zone)
// ─────────────────────────────────────────────────────────────────────────────
bgcolor(close < trendEma ? color.new(color.red, 92) : na)

// ─────────────────────────────────────────────────────────────────────────────
// Plots
// ─────────────────────────────────────────────────────────────────────────────
plot(trendEma, title="Trend EMA (100)", color=color.new(color.white, 60), linewidth=1)
