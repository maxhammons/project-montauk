//@version=6
strategy("Project Montauk 9.0 — Regime Score",
     overlay=true,
     shorttitle="Montauk 9.0-RS",
     initial_capital=1000,
     default_qty_type=strategy.percent_of_equity,
     default_qty_value=100,
     pyramiding=0,
     commission_type=strategy.commission.percent,
     commission_value=0,
     slippage=0,
     calc_on_every_tick=false,
     process_orders_on_close=true)

// ─────────────────────────────────────────────────────────────────────────────
// Group 1 — Indicators
// ─────────────────────────────────────────────────────────────────────────────
rsiLen          = input.int(7,     title="RSI Length",                   minval=1,   group="1 - Indicators")
maLen           = input.int(200,   title="Long EMA Length",             minval=1,   group="1 - Indicators")
volShort        = input.int(25,    title="Short Vol Window",            minval=1,   group="1 - Indicators")
volLong         = input.int(50,    title="Long Vol Window",             minval=1,   group="1 - Indicators")
ddLookback      = input.int(200,   title="Drawdown Lookback",          minval=1,   group="1 - Indicators")

// ─────────────────────────────────────────────────────────────────────────────
// Group 2 — Signal Weights
// ─────────────────────────────────────────────────────────────────────────────
wDrawdown       = input.float(0.1,  title="Weight: Drawdown",          step=0.05,  group="2 - Weights")
wRsi            = input.float(0.3,  title="Weight: RSI",               step=0.05,  group="2 - Weights")
wVol            = input.float(0.4,  title="Weight: Volatility",        step=0.05,  group="2 - Weights")
wPriceMa        = input.float(0.2,  title="Weight: Price/MA",          step=0.05,  group="2 - Weights")

// ─────────────────────────────────────────────────────────────────────────────
// Group 3 — Thresholds
// ─────────────────────────────────────────────────────────────────────────────
entryThresh     = input.float(0.55, title="Entry Threshold",           step=0.05,  group="3 - Thresholds")
exitThresh      = input.float(0.7,  title="Exit Threshold",            step=0.05,  group="3 - Thresholds")
panicRsi        = input.int(10,     title="Panic RSI (unconditional)",  minval=1,   group="3 - Thresholds")

// ─────────────────────────────────────────────────────────────────────────────
// Group 4 — Normalization
// ─────────────────────────────────────────────────────────────────────────────
ddCenter        = input.float(-35.0, title="Drawdown Center (%)",      step=5.0,   group="4 - Normalization")
ddScale         = input.float(15.0,  title="Drawdown Scale",           step=1.0,   group="4 - Normalization")
rsiCenter       = input.float(35.0,  title="RSI Center",               step=5.0,   group="4 - Normalization")
rsiScale        = input.float(15.0,  title="RSI Scale",                step=1.0,   group="4 - Normalization")
maCenter        = input.float(1.0,   title="Price/MA Center",          step=0.1,   group="4 - Normalization")
maScale         = input.float(0.2,   title="Price/MA Scale",           step=0.05,  group="4 - Normalization")

// ─────────────────────────────────────────────────────────────────────────────
// Indicator computation
// ─────────────────────────────────────────────────────────────────────────────
rsiVal          = ta.rsi(close, rsiLen)
maLong          = ta.ema(close, maLen)
peakHigh        = ta.highest(high, ddLookback)
troughLow       = ta.lowest(low, ddLookback)

// Realized volatility = stdev of daily returns over N bars
dailyReturn     = close / close[1] - 1.0
rVolShort       = ta.stdev(dailyReturn, volShort)
rVolLong        = ta.stdev(dailyReturn, volLong)

// ─────────────────────────────────────────────────────────────────────────────
// tanh normalization helper
// ─────────────────────────────────────────────────────────────────────────────
tanhNorm(float value, float center, float scale) =>
    scale == 0.0 ? 0.5 : (math.tanh((value - center) / scale) + 1.0) / 2.0

// ─────────────────────────────────────────────────────────────────────────────
// Raw signal values
// ─────────────────────────────────────────────────────────────────────────────
ddPct       = peakHigh > 0 ? (close - peakHigh) / peakHigh * 100.0 : 0.0
risePct     = troughLow > 0 ? (close - troughLow) / troughLow * 100.0 : 0.0
volRatio    = rVolLong > 0 ? rVolShort / rVolLong : 1.0
priceMa     = maLong > 0 ? close / maLong : 1.0

// ─────────────────────────────────────────────────────────────────────────────
// Entry score (high = bullish recovery)
//   Deep drawdown  → high score (invert: more negative dd = higher)
//   Low RSI        → high score (invert: lower RSI = higher)
//   Low vol ratio  → high score (invert: lower ratio = higher)
//   Price below MA → high score (invert: lower price/MA = higher)
// ─────────────────────────────────────────────────────────────────────────────
sDdEntry    = tanhNorm(-ddPct,    -ddCenter, ddScale)
sRsiEntry   = tanhNorm(-rsiVal,   -rsiCenter, rsiScale)
sVolEntry   = tanhNorm(-volRatio, -1.0, 0.3)
sMaEntry    = tanhNorm(-priceMa,  -maCenter, maScale)

entryScore  = wDrawdown * sDdEntry + wRsi * sRsiEntry + wVol * sVolEntry + wPriceMa * sMaEntry

// ─────────────────────────────────────────────────────────────────────────────
// Exit score (high = bearish exhaustion)
//   High rise from trough → high score
//   High RSI              → high score
//   High vol ratio        → high score (storm arriving)
//   Price above MA        → high score
// ─────────────────────────────────────────────────────────────────────────────
sRiseExit   = tanhNorm(risePct,   50.0, 30.0)
sRsiExit    = tanhNorm(rsiVal,    65.0, rsiScale)
sVolExit    = tanhNorm(volRatio,  1.2, 0.3)
sMaExit     = tanhNorm(priceMa,   maCenter, maScale)

exitScore   = wDrawdown * sRiseExit + wRsi * sRsiExit + wVol * sVolExit + wPriceMa * sMaExit

// ─────────────────────────────────────────────────────────────────────────────
// Entry / Exit signals — crossover on score thresholds
// ─────────────────────────────────────────────────────────────────────────────
enterSignal = ta.crossover(entryScore, entryThresh)
exitSignal  = ta.crossover(exitScore, exitThresh)
panicSignal = rsiVal < panicRsi

// ─────────────────────────────────────────────────────────────────────────────
// Strategy execution
// ─────────────────────────────────────────────────────────────────────────────
if enterSignal and strategy.position_size == 0
    strategy.entry("Long", strategy.long)

if strategy.position_size > 0 and exitSignal
    strategy.close("Long", comment="Score Exit")

if strategy.position_size > 0 and panicSignal
    strategy.close("Long", comment="RSI Panic")

// ─────────────────────────────────────────────────────────────────────────────
// Plot — score bands and MA for visual reference
// ─────────────────────────────────────────────────────────────────────────────
plot(maLong, "Long EMA", color=color.gray, linewidth=1)
plotshape(enterSignal and strategy.position_size == 0, title="Entry", location=location.belowbar, style=shape.triangleup,   color=color.new(color.green, 0), size=size.small)
plotshape(exitSignal  and strategy.position_size > 0,  title="Exit",  location=location.abovebar, style=shape.triangledown, color=color.new(color.red, 0),   size=size.small)
plotshape(panicSignal and strategy.position_size > 0,  title="Panic", location=location.abovebar, style=shape.xcross,       color=color.new(color.orange, 0), size=size.small)
